# AI Study Bot — Core Logic
@app.post("/chat")
async def chat(message: str, session_id: str):
    history = memory.get(session_id, [])
    history.append({"role": "user",
                    "content": message})
    response = groq.chat(
        model="llama-3.3-70b",
        messages=history
    )
    memory[session_id] = history  # persist
    return {"reply": response}
